Step 1 
Copy the files mod_authz_svn.so and mod_dav_svn.so from C:\Program Files\Subversion\httpd into C:\Program Files\Apache Software foundation\Apache2.2\modules . 

Step 2 
Modify the C:\Program Files\Apache Software foundation\Apache2.2\conf\httpd.conf file: 

Insert
Add the modules to the Apache server 
    LoadModule  dav_module             modules/mod_dav.so
    LoadModule  dav_svn_module         modules/mod_dav_svn.so

    <Location /svn>
      DAV svn
      SVNParentPath /projects/subversion
      SVNListParentPath on
      
      AuthzSVNAccessFile svn-auth-file.txt
      
      AuthType Digest
      AuthName "Subversion repository"
      #?? AuthDigestDomain /svn/
      AuthUserFile svn-auth-file
      Require valid-user
    </Location>
    CustomLog logs/svn_logfile "%t %u %{SVN-ACTION}e" env=SVN-ACTION


Step 3
create svn user access file for apache
    C:\Program Files\Apache Software Foundation\Apache2.2\bin>
    htpasswd -c ../svn-auth-file vvia  
    htpasswd ../svn-auth-file guest    (pw: guest, for read only)


Step 4
create svn access file  "svn-auth-file.txt"

    [/]
    vvia = rw
    guest = r
    * = r
