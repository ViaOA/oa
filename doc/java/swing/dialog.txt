    public JDialog getDialog() {
        if (dialog != null) return dialog;
        dialog = new JDialog(JfcDelegate.getDialogWindow(), DisplayName+" Search", ModalityType.APPLICATION_MODAL);
        
        dialog.setDefaultCloseOperation(JDialog.HIDE_ON_CLOSE);
        
        dialog.setLayout(new BorderLayout());
        dialog.add(getMainPanel(), BorderLayout.CENTER);
        dialog.pack();

        Component comp = dialog.getParent();
        if (comp != null) dialog.setLocationRelativeTo(comp);
        
        
        dialog.addWindowListener(new WindowAdapter() {
            @Override
            public void windowClosing(WindowEvent e) {
                if (!ItemSearchJfc.this.bSelected) {
                    ItemSearchJfc.this.onCancel();
                }
            }@Override
            public void windowOpened(WindowEvent e) {
                ItemSearchJfc.this.bSelected = false;
            }
        });
        return dialog;
    }

    private JDialog dlgUIContainer;
    public JDialog getUIContainerDialog() {
        if (dlgUIContainer != null) return dlgUIContainer;
        
        Window parent = SwingUtilities.getWindowAncestor(getUIPanel4());
        dlgUIContainer = new JDialog(parent, "UI Container", ModalityType.MODELESS);
        dlgUIContainer.setDefaultCloseOperation(JDialog.HIDE_ON_CLOSE);
        dlgUIContainer.setLayout(new BorderLayout());

        UIContainerJfc ui = new UIContainerJfc(hubObjectDef);
        JSplitPane sp = new JSplitPane(JSplitPane.HORIZONTAL_SPLIT, 
                new JScrollPane(ui.getTree()), ui.getEditPanel());
        dlgUIContainer.add(sp, BorderLayout.CENTER);

        dlgUIContainer.add(ui.getToolbar(), BorderLayout.NORTH);
        
        
        ui.getToolbar().getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0, false), "esc");
        ui.getToolbar().getActionMap().put("esc", new AbstractAction() {
            public void actionPerformed(ActionEvent e) {
                dlgUIContainer.setVisible(false);
            }
        });
        
        dlgUIContainer.pack();
        Dimension dimMax;
        if (parent != null) {
            dimMax = parent.getSize();
        }
        else {
             dimMax = Toolkit.getDefaultToolkit().getScreenSize();
             dimMax.width *= .9;
             dimMax.height *= .9;
        }
        Dimension dim = dlgUIContainer.getSize();
        dim.width = (int) Math.min(dim.width, dimMax.width);
        dim.height = (int) Math.min(dim.height, dimMax.height);
        dlgUIContainer.setSize(dim);
        
        if (parent != null) dlgUIContainer.setLocationRelativeTo(parent);
        return dlgUIContainer;
    }
