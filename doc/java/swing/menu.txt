    public OAMenuItem createDeleteMenuItem() {
        OAMenuItem miDelete = new OAMenuItem(getItems(), OAMenuItem.DELETE);
        miDelete.setText("Delete item ...");
        miDelete.setConfirmMessage("OK to Delete this VetPlan Item from system?");
        miDelete.setMnemonic(KeyEvent.VK_D);
        miDelete.setEnabled(ModelDelegate.getUserAccessHub(), UserAccess.PROPERTY_IsAdmin);
        miDelete.setUseSwingWorker(true);
        miDelete.setProcessingText("Processing ...", "Please wait while deleting Item");
        return miDelete;
    }
