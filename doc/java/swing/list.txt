    public OAList createList() {
        OAList lst = new OAList(getHub(), ItemType.PROPERTY_Name, 12, 23) {
            @Override
            public String getToolTipText(Object object, String defalutValue) {
                return super.getToolTipText(object, defalutValue);
            }
        };
        lst.setAllowDnD(true);
        lst.setAllowRemove(true);
        lst.setAllowDelete(false);
        lst.setAllowInsert(false);
        lst.setImageProperty(null);
        lst.setMaxImageHeight(14);
        lst.setMaxImageWidth(35);
        lst.setConfirmMessage("Ok to perform remove " +DisplayName+ "?");
        lst.setRemoveDragObject(true);
        lst.setIconColorProperty(null);
        lst.setBackgroundColorProperty(null);
        lst.setDoubleClickButton(null);
        // lst.setSelectionHub(getSelectItemTypes());
        lst.setBorder(new EmptyBorder(3,3,1,1));
        
        // lst.setEnabled(getUserAccessHub(), UserAccess.PROPERTY_IsAdmin);
        return lst;
