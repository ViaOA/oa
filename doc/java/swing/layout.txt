    private final String CARD_Splash = "splash";
    private final String CARD_ParentItem = "parentItem";
    private final String CARD_SubItem = "subItem";

    private JPanel cardPanel;
    private CardLayout cardLayout; 

    public JPanel getEditPanel() {
        if (cardPanel != null) return cardPanel;

        cardLayout = new CardLayout(5,5);
        cardPanel = new JPanel(cardLayout);

        cardPanel.add(new JLabel("VetPlan Items"), CARD_Splash);
        
        cardPanel.add(_getEditPanel(), CARD_ParentItem);
        cardPanel.add(getSubItemJfc()._getEditPanel(), CARD_SubItem);
        
        return cardPanel;
    }

    protected void onShowSubItemEditPanel() {
        cardLayout.show(getEditPanel(), CARD_SubItem);
    }

    public JPanel getEditPanel() {
        if (panel != null) return panel;
        panel = new JPanel(new GridBagLayout());
        panel.setBorder(new EmptyBorder(5,5, 3,3));
        
        GridBagConstraints gc = new GridBagConstraints();
        gc.insets = new Insets(1, 3, 1, 3);
        gc.anchor = gc.NORTHWEST;
        JLabel lbl;
        OATextField txt;
        OATextArea txta;
        
        lbl = new JLabel(DisplayName+" Name:");
        txt = createNameTextField();
        lbl.setLabelFor(txt);
        panel.add(lbl, gc);
        gc.gridwidth = gc.REMAINDER;
        panel.add(txt, gc);
        gc.gridwidth = 1;
        

        // auto-adjust textarea  <<< to take up a portion of extra room
        lbl = new JLabel("Description:");
        lbl.setLabelFor(getDescriptionTextArea());
        panel.add(lbl, gc);
        gc.weightx = gc.weighty = 1.0;
        gc.fill = gc.BOTH;
        panel.add(new JScrollPane(getDescriptionTextArea()), gc);
        gc.fill = gc.NONE;

        lbl = new JLabel("");
        gc.gridwidth = gc.REMAINDER;
        panel.add(lbl, gc);
        gc.gridwidth = 1;
        gc.weightx = gc.weighty = 0f;

        
        // take up remaining space  <<< 
        lbl = new JLabel("");
        gc.gridwidth = gc.REMAINDER;
        gc.weightx = gc.weighty = 1.0;
        panel.add(lbl, gc);
