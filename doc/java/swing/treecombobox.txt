



    public OATreeComboBox getTemplateTreeComboBox() {
        if (cboTemplate != null) return cboTemplate;

        TemplateSearchJfc searchJfc = new TemplateSearchJfc(getSpeciesJfc().getHub());
        
        searchJfc.getHub().setLinkHub(getItemSearchHub(), ItemSearch.PROPERTY_Template);
        
        OATree tree = searchJfc.getTree();
        tree.setPreferredSize(15, 40);
        
        cboTemplate = new OATreeComboBox(tree, getItemSearchHub(), OAString.cpp(ItemSearch.PROPERTY_Template, Template.PROPERTY_Name)) {
            public void onClear() {
                getItemSearch().setTemplate(null);
            }
        };
        cboTemplate.allowClearButton(true);
        return cboTemplate;
    }
