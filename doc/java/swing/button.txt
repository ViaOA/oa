


    public JButton getNewButton() {
        if (cmdNew == null) {

            URL url = OAButton.class.getResource("icons/New.gif");
            cmdNew = new JButton("New", new ImageIcon(url));

            cmdNew.addActionListener(new ActionListener() {
                @Override
                public void actionPerformed(ActionEvent e) {
                    onNew();
                }
            });
            new ButtonController(getHub(), cmdNew);
            OAButton.setup(cmdNew);
            cmdNew.setFocusPainted(true);
            

            new EnabledController(cmdNew, ModelDelegate.getUserAccessHub(), UserAccess.PROPERTY_IsUser);
        }
        return cmdNew;
    }
