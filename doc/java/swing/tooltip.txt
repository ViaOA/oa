


    public String getToolTipText(MouseEvent event) {
        String s = super.getToolTipText(event);
        if (event != null) {
            Point pt = event.getPoint();
            int row = rowAtPoint(pt);
            int col = columnAtPoint(pt);
            s = getToolTipText(row, col, s);
        }
        return s;
    }
