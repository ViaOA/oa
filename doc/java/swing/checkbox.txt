
    public OACheckBox createShowInactiveUserCheckBox() {
        OACheckBox chk = new OACheckBox(hubUserUI, UserUI.PROPERTY_ShowInactiveUsers);
        chk.setText("Show inactive Users");
        chk.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                JCheckBox chk = (JCheckBox) e.getSource();
                UserJfc.this.onShowInactiveActionPerformed(chk.isSelected());
            }
        });
        chk.setToolTipText("check this to include Users who are inactive.");
        return chk;
    }

    public OACheckBox createReasonCheckBox() {
        OACheckBox chk = new OACheckBox(getHub(), ExamItem.PROPERTY_ReasonForVisit);
        chk.setText("Reason for visit");
        chk.setToolTipText("Is the item the reason for the visit?");
        chk.setEnabled(ModelDelegate.getUserAccessHub(), UserAccess.PROPERTY_IsUser);
        chk.setEnabled(ModelDelegate.getUserAccessHub(), UserAccess.PROPERTY_ExamEnabled);
        return chk;
    }
