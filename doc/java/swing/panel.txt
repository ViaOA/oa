    public JPanel getEditPanel() {
        if (panel != null) return panel;
        panel = new JPanel(new GridBagLayout());
        panel.setBorder(new EmptyBorder(5,5, 3,3));

        gc.anchor = gc.NORTHWEST;

        OATextArea txta;

        lbl.setLabelFor(txt);
        panel.add(lbl, gc);
        gc.gridwidth = gc.REMAINDER;
        panel.add(txt, gc);
        gc.gridwidth = 1;

        // take up remaining space
        lbl = new JLabel("");
        gc.gridwidth = gc.REMAINDER;
        gc.weightx = gc.weighty = 1.0;
        panel.add(lbl, gc);
        
        
        
        JPanel pan = new JPanel(new BorderLayout());
        pan.add(panel, BorderLayout.NORTH);
        panel = pan;
        
        JTabbedPane tab = getTabbedPane();

        
        tab.addTab("Description", null, new JScrollPane(createDescriptionTextArea()), null);
        
        tab.addTab("Sub Categories", null, getSubItemCategoryPanel(), null);
        
        tab.addTab("Items", null, getItemPanel(), null);

        panel.add(tab, BorderLayout.CENTER);

        return panel;
    }
    protected OATextField getNameTextField() {
