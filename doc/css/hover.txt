>> Jquery way:
        
        $(table).find("tr").hover(
        function() { 
            $(this).addClass('selectedRow'); 
            $(this).contents('td').addClass('mouseOverRow');
        }, 
        function() { 
            $(this).removeClass('selectedRow'); 
            $(this).contents('td').removeClass('mouseOverRow'); 
        });
